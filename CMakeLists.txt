cmake_minimum_required(VERSION 3.0.2)
project(prod_cons_pkg)

## Compile as C++11, supported in ROS Kinetic and newer
# add_compile_options(-std=c++11)

## Find catkin macros and libraries
## if COMPONENTS list like find_package(catkin REQUIRED COMPONENTS xyz)
## is used, also find other catkin packages
find_package(catkin REQUIRED COMPONENTS
  roscpp
  std_msgs
)


catkin_package(
#  INCLUDE_DIRS include
#  LIBRARIES test_program
  CATKIN_DEPENDS roscpp std_msgs
)

add_executable(producer_node src/producer_node.cpp)
add_dependencies(producer_node ${producer_node_EXPORTED_TARGETS} ${catkin_EXPORTED_TARGETS})
target_link_libraries( producer_node ${catkin_LIBRARIES} )

add_executable(consumer_node src/consumer_node.cpp)
add_dependencies(consumer_node ${consumer_node_EXPORTED_TARGETS} ${catkin_EXPORTED_TARGETS})
target_link_libraries( consumer_node ${catkin_LIBRARIES} )


 install(TARGETS producer_node consumer_node
   RUNTIME DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
 )
 
 
## Mark other files for installation (e.g. launch and bag files, etc.)
install(DIRECTORY launch/
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/launch
  FILES_MATCHING
  PATTERN "*.launch"
)




include_directories(
# include
  ${catkin_INCLUDE_DIRS}
)
